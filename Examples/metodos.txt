//Instancias de un sensor, motor, relevador y servo
const boton = new Sensor(A5)
const motor1 = new Motor(5)
const relay1 = new Relay(6)
const servo1 = new Servo(7)

task main()
{
  while(true)
  {
    if(boton)
    {
      relay1.on
      servo1.position(180)
      motor.run(255)
    }
    else
    {
      relay1.off
      servo1.position(0)
      motor.run(0)
    }
  }
}
