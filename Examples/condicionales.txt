//Instancia de un objeto tipo Sensor
const sensorLuz = new Sensor(A5)

//Instancia de un relevador
const rele1 = new Relay(5)  

task main()
{
  while(true)
  {
    //Si el sensor detecta luz
    if(sensorLuz)
    {
      rele1.off
    }

    //Si el sensor no detecta luz
    else
    {
      rele1.on
    }
  }
}
